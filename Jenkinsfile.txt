pipeline {
    agent {
        docker {
            image 'maven:3.9-eclipse-temurin-17'
        }
    }

    stages {
        stage('Build') {
            steps {
                // اینجا می‌گیم برو داخل پوشه my-app و اونجا بیلد رو انجام بده
                dir('my-app') {
                    echo 'Building the application...'
                    sh 'mvn clean package'
                }
            }
        }

        stage('Test') {
            steps {
                dir('my-app') {
                    echo 'Running Tests...'
                    sh 'mvn test'
                }
            }
        }

        stage('Archive') {
            steps {
                // ذخیره کردن فایل ساخته شده برای دانلود
                archiveArtifacts artifacts: 'my-app/target/*.jar', fingerprint: true
            }
        }
    }
}
