pipeline {
    agent {
        docker {
            image 'maven:3.9-eclipse-temurin-17'
            // دیگر نیازی به دسترسی root برای دسترسی به فایل‌های سیستم اوبونتو نیست
            // چون فایل‌ها الان داخل محیط خودِ کانتینرِ جنکینز هستند
            args '' 
        }
    }

    stages {
        stage('Checkout') {
            steps {
                // این دستور جادویی است! 
                // جنکینز می‌رود سراغ آدرسی که در تنظیمات Job به او دادی 
                // و تمام کدها را در ورک‌اسپیس خودش Clone می‌کند.
                checkout scm
                echo '✅ Code successfully pulled from GitHub!'
            }
        }

        stage('Build & Test') {
            steps {
                // نکته طلایی: وقتی کد را از Git می‌گیری، 
                // ساختار پوشه‌ها دقیقا همان چیزی است که در GitHub هست.
                // اگر فایل pom.xml در ریشه (root) است، دیگر نیازی به dir('my-app') نیست.
                // اما اگر در GitHub هم داخل پوشه my-app است، همان را نگه می‌داریم.
                
                dir('my-app') { 
                    echo 'Building Maven project...'
                    sh 'mvn clean package -DskipTests' // اضافه کردن skipTests برای سرعت بیشتر در تست اول
                }
            }
        }

        stage('Archive Artifacts') {
            steps {
                // پیدا کردن فایل JAR که ساخته شده
                archiveArtifacts artifacts: 'my-app/target/*.jar', fingerprint: true
            }
        }
    }

    post {
        success {
            echo '🎉 YES! Build Successful, Lili! Your code is safe in GitHub and built perfectly!'
        }
        failure {
            echo '❌ Oh no! Something went wrong. Check the logs.'
        }
    }
}
