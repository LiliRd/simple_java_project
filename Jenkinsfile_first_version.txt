pipeline {
    agent {
        docker {
            image 'maven:3.9-eclipse-temurin-17'
            // این خط خیلی مهمه! اجازه می‌ده کانتینر به فایل‌های سیستم دسترسی داشته باشه
            args '-u root' 
        }
    }

    stages {
        stage('Checkout') {
            steps {
                // اگر از Git استفاده نمی‌کنی و فقط فایل‌ها رو دستی گذاشتی، 
                // این مرحله فعلاً فقط برای اطمینان از وجود فایل‌هاست.
                echo 'Checking files...'
                sh 'ls -R'
            }
        }

        stage('Build & Test') {
            steps {
                // جادوی اصلی اینجاست: ما اول وارد پوشه my-app می‌شیم
                dir('my-app') {
                    echo 'Building inside my-app directory...'
                    sh 'mvn clean package'
                }
            }
        }

        stage('Archive Artifacts') {
            steps {
                // حالا که بیلد تموم شد، فایل JAR رو از داخل پوشه my-app/target پیدا می‌کنیم
                archiveArtifacts artifacts: 'my-app/target/*.jar', fingerprint: true
            }
        }
    }

    post {
        success {
            echo '🎉 YES! Build Successful, Lili!'
        }
        failure {
            echo '❌ Oh no! Something went wrong. Check the logs.'
        }
    }
}
